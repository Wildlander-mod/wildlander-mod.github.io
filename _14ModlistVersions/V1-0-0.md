<script>
$(document).ready(function(){

let currentPopupRow = null;

function initializeFilters() {
  const table = document.querySelector('.modlist-table table');
  if (!table) return;
  
  const rows = Array.from(table.querySelectorAll('tbody tr'));
  const categories = new Set();
  const groups = new Set();
  
  rows.forEach(row => {
    const cells = row.querySelectorAll('td');
    if (cells.length >= 7) {
      const group = cells[0].textContent.trim();
      const category = cells[6].textContent.trim();
      if (group && group !== 'Group') groups.add(group);
      if (category && category !== 'Category') categories.add(category);
    }
  });
  
  const categoryFilter = document.getElementById('categoryFilter');
  const groupFilter = document.getElementById('groupFilter');
  
  Array.from(categories).sort().forEach(cat => {
    const option = document.createElement('option');
    option.value = cat;
    option.textContent = cat;
    categoryFilter.appendChild(option);
  });
  
  Array.from(groups).sort().forEach(grp => {
    const option = document.createElement('option');
    option.value = grp;
    option.textContent = grp;
    groupFilter.appendChild(option);
  });
  
  document.getElementById('modlistSearch').addEventListener('input', applyFilters);
  categoryFilter.addEventListener('change', applyFilters);
  groupFilter.addEventListener('change', applyFilters);
  
  updateResultCount();
}

function applyFilters() {
  const table = document.querySelector('.modlist-table table');
  if (!table) return;
  
  const searchTerm = document.getElementById('modlistSearch').value.toLowerCase();
  const categoryFilter = document.getElementById('categoryFilter').value;
  const groupFilter = document.getElementById('groupFilter').value;
  
  const rows = table.querySelectorAll('tbody tr');
  let visibleCount = 0;
  
  rows.forEach(row => {
    const cells = row.querySelectorAll('td');
    if (cells.length < 7) {
      row.style.display = 'none';
      return;
    }
    
    const group = cells[0].textContent.trim();
    const modName = cells[1].textContent.toLowerCase();
    const category = cells[6].textContent.trim();
    const author = cells[7].textContent.toLowerCase();
    
    const matchesSearch = !searchTerm || 
                         modName.includes(searchTerm) ||
                         author.includes(searchTerm) ||
                         category.toLowerCase().includes(searchTerm);
    
    const matchesCategory = !categoryFilter || category === categoryFilter;
    const matchesGroup = !groupFilter || group === groupFilter;
    
    const isVisible = matchesSearch && matchesCategory && matchesGroup;
    row.style.display = isVisible ? '' : 'none';
    
    if (isVisible) visibleCount++;
  });
  
  updateResultCount(visibleCount, rows.length);
}

function updateResultCount(visible, total) {
  const table = document.querySelector('.modlist-table table');
  if (!table) return;
  
  if (visible === undefined) {
    visible = Array.from(table.querySelectorAll('tbody tr')).filter(row => row.style.display !== 'none').length;
    total = table.querySelectorAll('tbody tr').length;
  }
  
  const countElement = document.getElementById('filterResultCount');
  if (countElement) {
    countElement.textContent = 'Showing ' + visible + ' of ' + total + ' mods';
  }
}

function clearAllFilters() {
  document.getElementById('modlistSearch').value = '';
  document.getElementById('categoryFilter').value = '';
  document.getElementById('groupFilter').value = '';
  applyFilters();
}

function initModlistPopups() {
  const table = document.querySelector('.modlist-table table');
  if (!table) {
    console.warn('Modlist table not found');
    return;
  }
  
  const rows = Array.from(table.querySelectorAll('tbody tr'));
  rows.forEach(row => {
    const modCell = row.querySelector('td:nth-child(2)');
    if (modCell) {
      modCell.addEventListener('click', (e) => handleModNameClick(e, row));
    }
  });
} 

function handleModNameClick(event, row) {
  if (currentPopupRow === row) {
    hideModlistPopup();
    currentPopupRow = null;
    return;
  }

  hideModlistPopup();
  showModlistPopup(event, row);
  currentPopupRow = row;
}

function showModlistPopup(event, row) {
  const cells = row.querySelectorAll('td');
  if (cells.length < 12) return;
  
  const data = {
    group: cells[0]?.textContent?.trim() || '',
    mo2Name: cells[1]?.textContent?.trim() || '',
    lookupUrl: cells[2]?.textContent?.trim() || '',
    version: cells[3]?.textContent?.trim() || '',
    nexusUrl: cells[4]?.textContent?.trim() || '',
    nexusModName: cells[5]?.textContent?.trim() || '',
    category: cells[6]?.textContent?.trim() || '',
    author: cells[7]?.textContent?.trim() || '',
    authorUrl: cells[8]?.textContent?.trim() || '',
    kofi: cells[9]?.textContent?.trim() || '',
    paypal: cells[10]?.textContent?.trim() || '',
    patreon: cells[11]?.textContent?.trim() || ''
  };
  
  let popup = document.getElementById('modlist-popup');
  if (!popup) {
    popup = document.createElement('div');
    popup.id = 'modlist-popup';
    popup.className = 'modlist-popup';
    document.body.appendChild(popup);
    window.addEventListener('scroll', hideModlistPopup);
  }
  
  popup.innerHTML = '<div><strong>MO2 Name:</strong> ' + data.mo2Name + '</div>' +
    '<div><strong>Group:</strong> ' + data.group + '</div>' +
    '<div><strong>Version:</strong> ' + (data.version || 'N/A') + '</div>' +
    '<div><strong>Category:</strong> ' + (data.category || 'N/A') + '</div>' +
    '<div><strong>Nexus Mod Name:</strong> ' + (data.nexusModName || 'N/A') + '</div>' +
    '<div><strong>Author:</strong> ' + (data.author || 'N/A') + '</div>' +
    (data.authorUrl ? '<div><strong>Author URL:</strong> <a href="' + data.authorUrl + '" target="_blank">Profile</a></div>' : '') +
    (data.nexusUrl ? '<div><strong>Nexus URL:</strong> <a href="' + data.nexusUrl + '" target="_blank">Mod Page</a></div>' : '') +
    '<div><strong>Ko-fi:</strong> ' + (data.kofi ? '<a href="' + data.kofi + '" target="_blank">Donate</a>' : 'No information') + '</div>' +
    '<div><strong>PayPal:</strong> ' + (data.paypal ? '<a href="' + data.paypal + '" target="_blank">Donate</a>' : 'No information') + '</div>' +
    '<div><strong>Patreon:</strong> ' + (data.patreon ? '<a href="' + data.patreon + '" target="_blank">Subscribe</a>' : 'No information') + '</div>';
  
  popup.style.display = 'block';
  popup.style.left = (event.clientX + 80) + 'px';
  popup.style.top = event.clientY + 'px';
}

function hideModlistPopup() {
  const popup = document.getElementById('modlist-popup');
  if (popup) {
    popup.style.display = 'none';
  }
  window.removeEventListener('scroll', hideModlistPopup);
}

initializeFilters();
initModlistPopups();

});
</script>
