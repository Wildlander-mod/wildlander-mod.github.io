function initializeFilters() {
  var table = document.querySelector('.modlist-table table');
  if (!table) return;
  
  var tbody = table.querySelector('tbody');
  if (!tbody) return;
  
  // Extract unique categories and groups
  var categories = new Set();
  var groups = new Set();
  
  Array.from(tbody.querySelectorAll('tr')).forEach(function(row) {
    var cells = row.querySelectorAll('td');
    if (cells.length >= 7) {
      var group = cells[0].textContent.trim();
      var category = cells[6].textContent.trim();
      if (group && group !== 'Group') groups.add(group);
      if (category && category !== 'Category') categories.add(category);
    }
  });
  
  // Populate category filter
  var categoryFilter = document.getElementById('categoryFilter');
  Array.from(categories).sort().forEach(function(cat) {
    var option = document.createElement('option');
    option.value = cat;
    option.textContent = cat;
    categoryFilter.appendChild(option);
  });
  
  // Populate group filter
  var groupFilter = document.getElementById('groupFilter');
  Array.from(groups).sort().forEach(function(grp) {
    var option = document.createElement('option');
    option.value = grp;
    option.textContent = grp;
    groupFilter.appendChild(option);
  });
  
  // Attach event listeners
  $('#modlistSearch').on('keyup', applyFilters);
  $(categoryFilter).on('change', applyFilters);
  $(groupFilter).on('change', applyFilters);
  $('#clearFilters').on('click', clearAllFilters);

  updateResultCount();

      modName.includes(searchTerm) || 
      author.includes(searchTerm) ||
      category.toLowerCase().includes(searchTerm);
    
    // Apply category filter
    var categoryMatch = !categoryFilter || category === categoryFilter;
    
    // Apply group filter
    var groupMatch = !groupFilter || group === groupFilter;
    
    var show = searchMatch && categoryMatch && groupMatch;
    row.style.display = show ? '' : 'none';
    
    if (show) visibleCount++;
  });
  
  updateResultCount(visibleCount, totalCount);
}

function clearAllFilters() {
  document.getElementById('modlistSearch').value = '';
  document.getElementById('categoryFilter').value = '';
  document.getElementById('groupFilter').value = '';
  applyFilters();
}

function updateResultCount(visible, total) {
  var counter = document.getElementById('filterResultCount');
  if (!counter) return;
  
  var table = document.querySelector('.modlist-table table tbody');
  if (!table) return;
  
  if (visible === undefined || total === undefined) {
    var allRows = table.querySelectorAll('tr').length;
    var visibleRows = Array.from(table.querySelectorAll('tr')).filter(function(r) { 
      return r.style.display !== 'none'; 
    }).length;
    visible = visibleRows;
    total = allRows;
  }
  
  counter.textContent = 'Showing ' + visible + ' of ' + total + ' mods';
}
