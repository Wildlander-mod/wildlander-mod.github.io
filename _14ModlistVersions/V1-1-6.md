## Complete V1.1.6 Modlist

Use the search box and filters below to find mods by:

- **Search** - Find mods by name or author
- **Category** - Filter by gameplay category
- **Group** - Filter by load order group

**Hover over MO2 Name** to see detailed information including version, Nexus mod name, and author details.

<script src="https://ajax.googleapis.com/ajax/libs/jquery/3.6.1/jquery.min.js"></script>
<script>
let modlistFilterTimeout;

$(document).ready(function(){
  initModlistFilters();
  initModlistTooltips();
});

function initModlistFilters() {
  const table = document.querySelector('.modlist-table table');
  if (!table) {
    console.warn('Modlist table not found');
    return;
  }
  
  const rows = Array.from(table.querySelectorAll('tbody tr'));
  const categories = new Set();
  const groups = new Set();
  
  rows.forEach(row => {
    const cells = row.querySelectorAll('td');
    if (cells.length >= 5) {
      categories.add(cells[4]?.textContent?.trim());
      groups.add(cells[0]?.textContent?.trim());
    }
  });
  
  const categoryFilter = document.getElementById('categoryFilter');
  const groupFilter = document.getElementById('groupFilter');
  
  Array.from(categories).sort().forEach(cat => {
    const option = document.createElement('option');
    option.value = cat;
    option.textContent = cat;
    categoryFilter.appendChild(option);
  });
  
  Array.from(groups).sort().forEach(grp => {
    const option = document.createElement('option');
    option.value = grp;
    option.textContent = grp;
    groupFilter.appendChild(option);
  });
  
  document.getElementById('modlistSearch').addEventListener('keyup', debounceModlistFilter);
  categoryFilter.addEventListener('change', filterModlistTable);
  groupFilter.addEventListener('change', filterModlistTable);
  
  updateModlistFilterCount(rows.length, rows.length);
}

function debounceModlistFilter() {
  clearTimeout(modlistFilterTimeout);
  modlistFilterTimeout = setTimeout(filterModlistTable, 300);
}

function initModlistTooltips() {
  const table = document.querySelector('.modlist-table table');
  if (!table) {
    console.warn('Modlist table not found');
    return;
  }
  
  const rows = Array.from(table.querySelectorAll('tbody tr'));
  rows.forEach(row => {
    const modCell = row.querySelector('td:nth-child(2)');
    if (!modCell) return;
    
    modCell.style.cursor = 'pointer';
    modCell.addEventListener('mouseenter', (e) => showModlistTooltip(e, row));
    modCell.addEventListener('mouseleave', hideModlistTooltip);
  });
}

function showModlistTooltip(event, row) {
  const cells = row.querySelectorAll('td');
  const data = {
    group: cells[0]?.textContent || '',
    mo2Name: cells[1]?.textContent || '',
    version: cells[2]?.textContent || '',
    nexusModName: cells[3]?.textContent || '',
    category: cells[4]?.textContent || '',
    author: cells[5]?.textContent || '',
    nexusUrl: cells[6]?.textContent || ''
  };
  
  let tooltip = document.getElementById('modlist-tooltip');
  if (!tooltip) {
    tooltip = document.createElement('div');
    tooltip.id = 'modlist-tooltip';
    tooltip.className = 'modlist-tooltip';
    document.body.appendChild(tooltip);
  }
  
  tooltip.innerHTML = `
    <div><strong>Group:</strong> ${data.group}</div>
    <div><strong>MO2 Name:</strong> ${data.mo2Name}</div>
    <div><strong>Version:</strong> ${data.version || 'N/A'}</div>
    <div><strong>Nexus Mod Name:</strong> ${data.nexusModName || 'N/A'}</div>
    <div><strong>Category:</strong> ${data.category || 'N/A'}</div>
    <div><strong>Author:</strong> ${data.author || 'N/A'}</div>
    <div><strong>Nexus URL:</strong> ${data.nexusUrl ? '<a href="' + data.nexusUrl + '" target="_blank">View on Nexus</a>' : 'N/A'}</div>
  `;
  tooltip.style.display = 'block';
  tooltip.style.left = event.pageX + 10 + 'px';
  tooltip.style.top = event.pageY + 10 + 'px';
}

function hideModlistTooltip() {
  const tooltip = document.getElementById('modlist-tooltip');
  if (tooltip) tooltip.style.display = 'none';
}

function filterModlistTable() {
  const searchTerm = document.getElementById('modlistSearch').value.toLowerCase();
  const categoryFilter = document.getElementById('categoryFilter').value;
  const groupFilter = document.getElementById('groupFilter').value;
  
  const table = document.querySelector('.modlist-table table');
  const rows = Array.from(table.querySelectorAll('tbody tr'));
  
  let visibleCount = 0;
  rows.forEach(row => {
    const cells = row.querySelectorAll('td');
    const mo2Name = cells[1]?.textContent?.toLowerCase() || '';
    const author = cells[5]?.textContent?.toLowerCase() || '';
    const category = cells[4]?.textContent?.trim() || '';
    const group = cells[0]?.textContent?.trim() || '';
    
    const matchesSearch = mo2Name.includes(searchTerm) || author.includes(searchTerm);
    const matchesCategory = !categoryFilter || category === categoryFilter;
    const matchesGroup = !groupFilter || group === groupFilter;
    
    const isVisible = matchesSearch && matchesCategory && matchesGroup;
    row.style.display = isVisible ? '' : 'none';
    if (isVisible) visibleCount++;
  });
  
  updateModlistFilterCount(visibleCount, rows.length);
}

function updateModlistFilterCount(visible, total) {
  const counter = document.getElementById('filterResultCount');
  counter.textContent = `Showing ${visible} of ${total} mods`;
}

function clearModlistFilters() {
  document.getElementById('modlistSearch').value = '';
  document.getElementById('categoryFilter').value = '';
  document.getElementById('groupFilter').value = '';
  filterModlistTable();
}

</script>

    <option value="">All Groups</option>
  </select>
  <button id="clearFilters" onclick="clearModlistFilters()">Clear Filters</button>
